def modular_pow(base, exponent, modulus):
    # Optimized bitwise approach
    res = 1
    base = base % modulus
    while exponent > 0:
        if (exponent % 2 == 1):
            res = (res * base) % modulus
        exponent = exponent // 2
        base = (base * base) % modulus
    return res
